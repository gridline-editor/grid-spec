% #1 base word

\newcommand{\introref}[2]{%
	\expandafter\gdef\csname #1_intro\endcsname{\textit{#1#2}}%
    \expandafter\gdef\csname #1_ref\endcsname{1}%
	\textit{#1#2}%
}

% #1 base word
% #2 inflection
\newcommand{\reftext}[2]{%
    \ifcsname #1_ref\endcsname
        \csname #1_intro\endcsname%
    \else
        \textbf{\textcolor{red}{[Reference ``#1`` not yet introduced]}}%
    \fi
}
